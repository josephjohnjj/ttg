82  static int static_unpack_msg(parsec_comm_engine_t *ce, uint64_t tag, void *data, long unsigned int size, int src_rank,
83  void *obj) {
84  static_set_arg_fct_type static_set_arg_fct;
85  int rank;
86  parsec_taskpool_t *tp = NULL;
87  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
88  msg_header_t *msg = static_cast<msg_header_t *>(data);
89  uint64_t op_id = msg->op_id;
90  tp = parsec_taskpool_lookup(msg->taskpool_id);
91  assert(NULL != tp);
92  static_map_mutex.lock();
93  try {
94  auto op_pair = static_id_to_op_map.at(op_id);
95  static_map_mutex.unlock();
96  tp->tdm.module->incoming_message_start(tp, src_rank, NULL, NULL, 0, NULL);
97  static_set_arg_fct = op_pair.first;
98  static_set_arg_fct(data, size, op_pair.second);
99  tp->tdm.module->incoming_message_end(tp, NULL);
100  return 0;
101  } catch (const std::out_of_range &e) {
102  void *data_cpy = malloc(size);
103  assert(data_cpy != 0);
104  memcpy(data_cpy, data, size);
105  if (ttg::tracing()) {
106  ttg::print("ttg_parsec(", rank, ") Delaying delivery of message (", src_rank, ", ", op_id, ", ", data_cpy, ", ",
107  size, ")");
108  }
109  delayed_unpack_actions.insert(std::make_pair(op_id, std::make_tuple(src_rank, data_cpy, size)));
110  static_map_mutex.unlock();
111  return 1;
112  }

289  inline uint64_t parsec_tasks_hash_fct(parsec_key_t key, int nb_bits, void *data) {
290  /* Use all the bits of the 64 bits key, project on the lowest base bits (0 <= hash < 1024) */
291  int b = 0;
292  uint64_t mask = ~0ULL >> (64 - nb_bits);
293  uint64_t h = (uint64_t)key;
294  (void)data;
295  while (b < 64) {
296  b += nb_bits;
297  h ^= (uint64_t)key >> b;
298  }
299  return (uint64_t)(h & mask);
300  }
